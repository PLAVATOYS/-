  <h2 id="result"></h2>

  <script>
    let prizes = ["A賞", "B賞", "C賞", "D賞", "D賞"];
    function draw() {
      if (prizes.length === 0) {
        document.getElementById("result").innerText = "全部抽完囉！";
        return;
      }
      let index = Math.floor(Math.random() * prizes.length);
      let prize = prizes.splice(index, 1)[0];
      document.getElementById("result").innerText = `你抽中了：${prize}`;
    }
  </script>
